rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Users - require auth
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null && request.auth.uid == userId;
    }

    // Addresses - require auth
    match /addresses/{addressId} {
      allow read, write: if request.auth != null;
    }

    // Products - public read, auth for write, but allow stock updates for payment
    match /products/{productId} {
      allow read: if true;
      allow create, update, delete: if request.auth != null;
      // Allow stock updates for payment processing (no auth required)
      allow update: if true;
    }

    // ORDERS - completely public for payment processing
    match /orders/{orderId} {
      allow read: if true;
      allow update: if true; // Allow all updates for payment processing
      allow create: if request.auth != null; // Still require auth to create orders
    }

    // PHARMACIES - completely public for payment processing
    match /pharmacies/{pharmacyId} {
      allow read: if true;
      allow update: if true; // Allow all updates for payment processing
      allow create: if request.auth != null;
    }

    // TRANSACTIONS - completely public for payment processing
    match /transactions/{transactionId} {
      allow read: if true;
      allow create: if true; // Allow payment processing to create transactions
      allow update: if request.auth != null;
    }

    // Settings - require auth for admin
    match /settings/{settingId} {
      allow read: if true;
      allow write: if request.auth != null &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // Withdrawals - require auth
    match /withdrawalRequests/{requestId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update: if request.auth != null &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // Admin full access
    match /{document=**} {
      allow read, write: if request.auth != null &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
  }
}
